#include <iostream>
using namespace std;

int main() {
    int n;

    // Ввод числа
    cout << "Введите натуральное число n (n > 9): ";
    cin >> n;

    if (n <= 9) {
        cout << "Ошибка: число должно быть больше 9." << endl;
        return 1; // Выход из программы с кодом ошибки
    }

    // Находим количество единиц
    int units = n % 10;

    // Находим количество десятков
    int tens = (n / 10) % 10;

    // Вывод результатов
    cout << "Число единиц: " << units << endl;
    cout << "Число десятков: " << tens << endl;

    return 0;
}
