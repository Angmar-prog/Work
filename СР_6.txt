#include <iostream>
#include <vector>

using namespace std;

int main() {
  int A, B;
  cout << "Введите начальную высоту улитки (A): ";
  cin >> A;
  cout << "Введите высоту дерева (B): ";
  cin >> B;

  // Массив погоды (0 - пасмурно, 1 - солнечно)
  vector<int> weather(30);
  cout << "Введите информацию о погоде за 30 дней (0 - пасмурно, 1 - солнечно):\n";
  for (int i = 0; i < 30; i++) {
    cin >> weather[i];
  }

  int current_height = A; // Текущая высота улитки

  // Перебор дней наблюдения
  for (int i = 0; i < 30; i++) {
    if (weather[i] == 1) { // Солнечный день
      current_height += 2;
    } else { // Пасмурный день
      current_height -= 1;
    }

    // Проверка, не достигла ли улитка верхушки дерева
    if (current_height >= B) {
      cout << "Улитка достигла верхушки дерева на " << i + 1 << "-й день.\n";
      return 0; // Завершаем программу, если улитка достигла верхушки
    }

    // Проверка, не упала ли улитка с дерева
    if (current_height < 0) {
      cout << "Улитка упала с дерева на " << i + 1 << "-й день.\n";
      return 0; // Завершаем программу, если улитка упала
    }
  }

  cout << "К концу 30-го дня улитка находилась на высоте " << current_height << " см.\n";

  return 0;
}
